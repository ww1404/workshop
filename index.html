", unsafe_type_allowed=True) service_name = st.text_input("اسم الخدمة الحالية *", help="الاسم الشائع للخدمة المقدمة") owning_unit = st.text_input("الوحدة التنظيمية المعنية بتقديم الخدمة *", help="الإدارة أو القسم المسؤول مباشرة عن تشغيل الخدمة") challenge_desc = st.text_area("رصد وتوصيف التحدي الحالي *", help="صِف العقبة الأساسية التي تواجه تقديم الخدمة (مثل: مراجعات ورقية، تكرار خطوات، تأخر اعتمادات)") # 3. فرص التحسين st.markdown("

📈 مؤشرات الأداء الحالية للورشة

", unsafe_type_allowed=True) # حساب بعض القيم الإحصائية السريعة total_opportunities = len(df) quick_wins_count = len(df[df["العائد المتوقع (محتسب تلقائياً)"].str.contains("أولوية قصوى")]) quick_wins_pct = int((quick_wins_count / total_opportunities) * 100) if total_opportunities > 0 else 0 most_active_table = df["رقم الطاولة"].mode()[0] if not df.empty else "لا يوجد" most_active_unit = df["الوحدة التنظيمية المعنية"].mode()[0] if not df.empty else "لا يوجد" m1, m2, m3, m4 = st.columns(4) with m1: st.metric(label="إجمالي الفرص المرصودة", value=f"{total_opportunities} فرصة") with m2: st.metric(label="فرص ذات أثر وعائد عالي جداً (Quick Wins)", value=f"{quick_wins_count} فرصة", delta=f"{quick_wins_pct}% من المحفظة") with m3: st.metric(label="الطاولة الأكثر تفاعلاً بالورشة", value=most_active_table) with m4: st.metric(label="أكثر وحدة تنظيمية مرصودة", value=most_active_unit) st.markdown("---") # 2. التحليلات والرسوم البيانية التفاعلية col_chart1, col_chart2 = st.columns(2) with col_chart1: st.subheader("🎯 توزيع العائد على الاستثمار المتوقع (محتسب تلقائياً)") roi_counts = df["العائد المتوقع (محتسب تلقائياً)"].value_counts() st.bar_chart(roi_counts, color="#007bff") st.caption("يعبر هذا المخطط عن أولوية التنفيذ والوفر المتوقع (الزمني والمالي) بناءً على تقييم المدخلات.") with col_chart2: st.subheader("👥 عدد الأفكار والفرص المقدمة من كل طاولة") table_counts = df["رقم الطاولة"].value_counts() st.bar_chart(table_counts, color="#28a745") st.caption("مستوى تفاعل طاولات ورشة العمل ومعدل تقديم الأفكار من كل طاولة.") st.markdown("---") # 3. جدول المحفظة الشامل وإمكانية الفلترة السريعة st.markdown("
